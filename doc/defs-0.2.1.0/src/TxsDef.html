-- TorXakis Interal Data Type Definitions:

module TxsDef
where

import CnectDef
import CstrDef
import FuncDef
import MapperDef
import ModelDef
import ProcDef
import PurpDef
import SortDef

-- torxakis definitions

data  TxsDef        =  DefSort      SortDef
                     | DefCstr      CstrDef
                     | DefFunc      FuncDef
                     | DefProc      ProcDef
                     | DefModel     ModelDef
                     | DefPurp      PurpDef
                     | DefMapper    MapperDef
                     | DefCnect     CnectDef
                     | DefChan
                     | DefVar
                     | DefStat
                     | DefGoal
     deriving (Eq,Ord,Read,Show)
